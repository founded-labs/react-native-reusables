### Usage

<PortalInfoAlert />

```tsx
import {
  ContextMenu,
  ContextMenuContent,
  ContextMenuItem,
  ContextMenuTrigger,
} from '@/components/ui/context-menu';
import { Text } from '@/components/ui/text';

```

```tsx
<ContextMenu>
  <ContextMenuTrigger>
    <Text>Right click</Text>
  </ContextMenuTrigger>
  <ContextMenuContent>
    <ContextMenuItem>
      <Text>Profile</Text>
    </ContextMenuItem>
    <ContextMenuItem>
      <Text>Billing</Text>
    </ContextMenuItem>
    <ContextMenuItem>
      <Text>Team</Text>
    </ContextMenuItem>
    <ContextMenuItem>
      <Text>Subscription</Text>
    </ContextMenuItem>
  </ContextMenuContent>
</ContextMenu>
```
