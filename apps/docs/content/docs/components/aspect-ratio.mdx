### Usage

```tsx
import { AspectRatio } from '@/components/ui/aspect-ratio';
import { Text } from '@/components/ui/text';
```

```tsx
<AspectRatio ratio={16 / 9} className='h-full w-full justify-center items-center'>
  <Text>16:9</Text>
</AspectRatio>
```
