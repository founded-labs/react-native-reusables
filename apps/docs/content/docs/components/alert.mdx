### Usage

```tsx
import { Alert, AlertDescription, AlertTitle } from '@/components/ui/alert';
import { Terminal } from '@/lib/icons/Terminal';
```

```tsx
<Alert icon={Terminal}>
  <AlertTitle>Heads up!</AlertTitle>
  <AlertDescription>
    You can use a terminal to run commands on your computer.
  </AlertDescription>
</Alert>
```
